domain_lower = chosen.lower()

zip_meta_old = None
zip_meta_new = None
if chosen == "MCO":
    zip_meta_old = parse_zip_name_mco(working_zip_name)
elif chosen == "SMR":
    zip_meta_old = parse_zip_name_smr(working_zip_name)
else:
    zip_meta_old = parse_zip_name_psy(working_zip_name)

zip_meta_new = parse_zip_name_newstyle(working_zip_name)

site3_for_name = None
yyyymm_for_name = None

if zip_meta_old:
    site3_for_name = zip_meta_old["site3"]
    yyyymm_for_name = zip_meta_old["yyyymm"]
elif zip_meta_new and zip_meta_new["domain"] == chosen:
    yyyymm_for_name = zip_meta_new["yyyymm"]
    site5_from_finess = FINESS_TO_SITE5.get(zip_meta_new["finess"])
    if site5_from_finess:
        site3_for_name = site5_to_site3(site5_from_finess)

if site3_for_name and yyyymm_for_name:
    out_name = f"{domain_lower}_{site3_for_name}_{yyyymm_for_name}_out.zip"
else:
    out_name = working_zip_name.replace("_in.zip", "_out.zip")
    if out_name == working_zip_name:
        out_name = (
            working_zip_name[:-4] + "_out.zip"
            if working_zip_name.lower().endswith(".zip")
            else working_zip_name + "_out.zip"
        )
