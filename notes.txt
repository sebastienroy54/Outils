SELECT COUNT(DISTINCT RSM.NOIP) AS NB_PATIENTS
FROM RSM
WHERE RSM.UE = '979'
  AND RSM.D8EEUE >= TO_DATE('01/01/2024','DD/MM/YYYY')
  AND RSM.D8EEUE <  TO_DATE('01/01/2025','DD/MM/YYYY')
  AND (
       RSM.CDDNC9 LIKE 'Z451%'
       OR RSM.CDRELIE LIKE 'Z451%'
       OR EXISTS (
            SELECT 1
            FROM DGN
            WHERE DGN.KYRES = RSM.KYRES
              AND DGN.KYCDDG LIKE 'Z451%'
       )
  );
