# -*- coding: utf-8 -*-
import pandas as pd

EXCEL_PATH = "Extraction DREAM Janvier 2026.xlsx"
CSV_PATH = "SEJOURS_Z515_JANVIER_2026.csv"

OUTPUT_MISSING = "IPP_absents_du_CSV.xlsx"
OUTPUT_COMMON = "IPP_communs_Excel_CSV.xlsx"

COL_IPP_EXCEL = "Patient::IP_NIP"
COL_CADRE_EXCEL = "I_Cadre Intervention"
COL_DATE = "I_Date Intervention"
COL_IPP_CSV = "IPP"

def read_csv_flexible(path):
    encodings = ["utf-8-sig", "utf-8", "cp1252", "latin1"]
    seps = [";", ","]
    last_err = None

    for enc in encodings:
        for sep in seps:
            try:
                df = pd.read_csv(path, sep=sep, encoding=enc, dtype=str)
                if df.shape[1] <= 1:
                    continue
                return df
            except Exception as e:
                last_err = e

    raise RuntimeError(f"Impossible de lire le CSV correctement: {last_err}")

def normalize_id(series):
    s = series.astype(str).str.strip()
    s = s.replace({"nan": None, "None": None, "": None})
    s = s.str.replace(r"\.0$", "", regex=True)
    return s

def clean_columns(df):
    df = df.copy()
    df.columns = [str(c).strip() for c in df.columns]
    return df

def main():

    # --- Lecture Excel
    df_x = pd.read_excel(EXCEL_PATH)
    df_x = clean_columns(df_x)

    # --- Conversion de la date
    if COL_DATE in df_x.columns:
        df_x[COL_DATE] = pd.to_datetime(df_x[COL_DATE], errors="coerce")
        df_x[COL_DATE] = df_x[COL_DATE].dt.strftime("%d/%m/%Y")

    # --- Filtre TéléCS
    df_x[COL_CADRE_EXCEL] = df_x[COL_CADRE_EXCEL].astype(str).str.strip()
    df_x_filtered = df_x[df_x[COL_CADRE_EXCEL] != "TéléCS"].copy()

    # --- Lecture CSV
    df_c = read_csv_flexible(CSV_PATH)
    df_c = clean_columns(df_c)

    # --- Normalisation IPP
    df_x_filtered[COL_IPP_EXCEL] = normalize_id(df_x_filtered[COL_IPP_EXCEL])
    df_c[COL_IPP_CSV] = normalize_id(df_c[COL_IPP_CSV])

    ipp_excel = set(df_x_filtered[COL_IPP_EXCEL].dropna().unique())
    ipp_csv = set(df_c[COL_IPP_CSV].dropna().unique())

    ipp_missing_in_csv = ipp_excel - ipp_csv
    ipp_common = ipp_excel & ipp_csv

    # --- Résultats
    df_missing = df_x_filtered[df_x_filtered[COL_IPP_EXCEL].isin(ipp_missing_in_csv)].copy()
    df_common = df_x_filtered[df_x_filtered[COL_IPP_EXCEL].isin(ipp_common)].copy()

    df_missing = df_missing.sort_values(by=[COL_IPP_EXCEL])
    df_common = df_common.sort_values(by=[COL_IPP_EXCEL])

    # --- Export Excel
    with pd.ExcelWriter(OUTPUT_MISSING, engine="openpyxl") as w:
        df_missing.to_excel(w, index=False, sheet_name="Excel_not_in_CSV")

    with pd.ExcelWriter(OUTPUT_COMMON, engine="openpyxl") as w:
        df_common.to_excel(w, index=False, sheet_name="Excel_and_CSV")

    print("Exports terminés")
    print(f"{OUTPUT_MISSING} : {len(df_missing)} lignes")
    print(f"{OUTPUT_COMMON} : {len(df_common)} lignes")

if __name__ == "__main__":
    main()
