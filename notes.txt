SELECT
    cadre_intervention,
    COUNT(*) AS patients
FROM (
    SELECT DISTINCT ON (id_patient)
        id_patient,
        cadre_intervention,
        date_intervention
    FROM soins_pall
    WHERE date_intervention >= '2016-01-01'
    ORDER BY
        id_patient,
        date_intervention,
        id_soins_pall
) t
GROUP BY cadre_intervention
ORDER BY cadre_intervention;
