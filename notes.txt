# ---- FORMAT DES DATES POUR LE FICHIER FINAL ----

# Date de pose : dd/mm/yyyy (sans heure)
if "Date de pose" in df_out.columns:
    d = pd.to_datetime(df_out["Date de pose"], errors="coerce", dayfirst=True)
    df_out["Date de pose"] = d.dt.strftime("%d/%m/%Y")

# Date saisie SEDITRACE : dd/mm/yyyy HH:MM:SS (avec heure)
if "Date saisie SEDITRACE" in df_out.columns:
    d = pd.to_datetime(df_out["Date saisie SEDITRACE"], errors="coerce", dayfirst=True)
    df_out["Date saisie SEDITRACE"] = d.dt.strftime("%d/%m/%Y %H:%M:%S")
