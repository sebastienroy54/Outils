SELECT
    S.KYNODA AS NDA,
    S.NMMAL  AS NOM,
    S.NMPMAL AS PRENOM,
    MIN(R.D8EEUE) AS DATE_ENTREE,
    MAX(R.D8SOUE) AS DATE_SORTIE
FROM
    SDO S,
    RSM R
WHERE
    S.KYNODA = R.NODA
    AND S.KYNODA IN (
        /* === COLLE ICI TA LISTE DE NDA === */
        123456,
        234567,
        345678
    )
GROUP BY
    S.KYNODA, S.NMMAL, S.NMPMAL
ORDER BY
    S.KYNODA;
