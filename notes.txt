SELECT
    ACS.NOUFCO,
    COUNT(DISTINCT ACS.NOACS) AS NB_CONSULTATIONS,
    COUNT(DISTINCT ACS.NOIP) AS NB_PATIENTS,
    MIN(ACS.DACO) AS PREMIERE_DATE,
    MAX(ACS.DACO) AS DERNIERE_DATE
FROM ACS
WHERE ACS.CDPHCO = 'MDA MORO'
  AND ACS.TYMAJ <> 'D'
GROUP BY
    ACS.NOUFCO
ORDER BY
    COUNT(DISTINCT ACS.NOACS) DESC, ACS.NOUFCO
;
