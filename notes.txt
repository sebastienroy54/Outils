import pandas as pd
from sqlalchemy import text

from connexion_bases import get_dimbdd_engine, get_dimbdd_config


def get_schema() -> str:
    cfg = get_dimbdd_config()
    return cfg.get("schema") or "public"


def get_engine():
    return get_dimbdd_engine()


def chercher_tables_avec_commentaires() -> pd.DataFrame:
    """
    Liste les tables du schéma qui ont un commentaire PostgreSQL non vide.
    """
    engine = get_engine()
    schema = get_schema()

    query = text("""
        SELECT
            n.nspname AS schema_name,
            c.relname AS table_name,
            obj_description(c.oid, 'pg_class') AS table_description
        FROM pg_class c
        JOIN pg_namespace n
            ON n.oid = c.relnamespace
        WHERE c.relkind = 'r'
          AND n.nspname = :schema
          AND obj_description(c.oid, 'pg_class') IS NOT NULL
          AND obj_description(c.oid, 'pg_class') <> ''
        ORDER BY c.relname
    """)

    with engine.connect() as conn:
        return pd.read_sql(query, conn, params={"schema": schema})


def chercher_colonnes_avec_commentaires() -> pd.DataFrame:
    """
    Liste les colonnes qui ont un commentaire PostgreSQL non vide.
    """
    engine = get_engine()
    schema = get_schema()

    query = text("""
        SELECT
            cols.table_schema,
            cols.table_name,
            cols.ordinal_position,
            cols.column_name,
            cols.data_type,
            pgd.description AS column_description
        FROM information_schema.columns cols
        LEFT JOIN pg_catalog.pg_statio_all_tables st
            ON st.schemaname = cols.table_schema
           AND st.relname = cols.table_name
        LEFT JOIN pg_catalog.pg_description pgd
            ON pgd.objoid = st.relid
           AND pgd.objsubid = cols.ordinal_position
        WHERE cols.table_schema = :schema
          AND pgd.description IS NOT NULL
          AND pgd.description <> ''
        ORDER BY cols.table_name, cols.ordinal_position
    """)

    with engine.connect() as conn:
        return pd.read_sql(query, conn, params={"schema": schema})


def chercher_tables_proches(motif: str) -> pd.DataFrame:
    """
    Cherche les tables dont le nom ressemble à un motif.
    """
    engine = get_engine()
    schema = get_schema()

    query = text("""
        SELECT
            table_schema,
            table_name
        FROM information_schema.tables
        WHERE table_schema = :schema
          AND table_type = 'BASE TABLE'
          AND lower(table_name) LIKE lower(:motif)
        ORDER BY table_name
    """)

    with engine.connect() as conn:
        return pd.read_sql(query, conn, params={"schema": schema, "motif": f"%{motif}%"})


def table_existe(table_name: str) -> bool:
    """
    Vérifie si une table existe dans le schéma.
    """
    engine = get_engine()
    schema = get_schema()

    query = text("""
        SELECT EXISTS (
            SELECT 1
            FROM information_schema.tables
            WHERE table_schema = :schema
              AND table_name = :table_name
        ) AS existe
    """)

    with engine.connect() as conn:
        result = conn.execute(query, {"schema": schema, "table_name": table_name}).scalar()
        return bool(result)


def tables_avec_colonne(column_name: str) -> pd.DataFrame:
    """
    Liste les tables qui contiennent une colonne donnée.
    """
    engine = get_engine()
    schema = get_schema()

    query = text("""
        SELECT
            table_schema,
            table_name,
            column_name,
            data_type
        FROM information_schema.columns
        WHERE table_schema = :schema
          AND lower(column_name) = lower(:column_name)
        ORDER BY table_name
    """)

    with engine.connect() as conn:
        return pd.read_sql(query, conn, params={"schema": schema, "column_name": column_name})


def chercher_finess_dans_tables_nofiness(finess: str = "750100166", max_tables: int | None = None) -> pd.DataFrame:
    """
    Cherche si un FINESS donné existe dans toutes les tables contenant la colonne 'nofiness'.
    Renvoie uniquement les tables où au moins une ligne existe.
    """
    engine = get_engine()
    schema = get_schema()

    df_tables = tables_avec_colonne("nofiness")

    if max_tables is not None:
        df_tables = df_tables.head(max_tables).copy()

    resultats = []

    with engine.connect() as conn:
        for table_name in df_tables["table_name"]:
            print(f"Recherche FINESS dans {table_name}...")

            full_table_name = f'"{schema}"."{table_name}"'
            query = text(f"""
                SELECT COUNT(*) AS nb
                FROM {full_table_name}
                WHERE CAST(nofiness AS TEXT) = :finess
            """)

            try:
                nb = conn.execute(query, {"finess": finess}).scalar()
                if nb and int(nb) > 0:
                    resultats.append({
                        "table_name": table_name,
                        "nofiness_recherche": finess,
                        "nb_lignes": int(nb),
                    })
            except Exception as e:
                resultats.append({
                    "table_name": table_name,
                    "nofiness_recherche": finess,
                    "nb_lignes": None,
                    "erreur": str(e),
                })

    return pd.DataFrame(resultats)


def apercu_finess(table_name: str, finess: str = "750100166", limit: int = 20) -> pd.DataFrame:
    """
    Retourne un aperçu des lignes d'une table pour un FINESS donné.
    """
    engine = get_engine()
    schema = get_schema()

    full_table_name = f'"{schema}"."{table_name}"'

    query = text(f"""
        SELECT *
        FROM {full_table_name}
        WHERE CAST(nofiness AS TEXT) = :finess
        LIMIT :limit
    """)

    with engine.connect() as conn:
        return pd.read_sql(query, conn, params={"finess": finess, "limit": limit})
