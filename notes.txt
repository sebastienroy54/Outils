SELECT DISTINCT
       RSM.NOIP AS IPP
FROM
       RSM
WHERE
       RSM.CDURM_P IN ('010','010S')
       AND RSM.D8EEUE >= TO_DATE('01/01/2023','DD/MM/YYYY')
       AND RSM.D8EEUE <  TO_DATE('01/01/2025','DD/MM/YYYY')
ORDER BY
       RSM.NOIP;
