SELECT COUNT(DISTINCT RSM.NOIP) AS FILE_ACTIVE_Z448
FROM RSM, RUM, RSS
WHERE RUM.NORES = RSM.KYRES
  AND RSS.KYRSS = RUM.KYRSS
  AND RSM.CDURM_P = '071S'
  AND RSS.D8DEB >= TO_DATE('01/01/2025','DD/MM/YYYY')
  AND RSS.D8DEB <  TO_DATE('01/01/2026','DD/MM/YYYY')
  AND (
       RSM.CDDNC9 LIKE 'Z448%'
       OR RSM.CDRELIE LIKE 'Z448%'
       OR EXISTS (
            SELECT 1
            FROM DGN
            WHERE DGN.KYRES = RSM.KYRES
              AND DGN.KYCDDG LIKE 'Z448%'
       )
  );




SELECT CASE
         WHEN TRUNC(MONTHS_BETWEEN(RSS.D8DEB, SDO.DANAIS)/12) BETWEEN 15 AND 24 THEN '15_24'
         WHEN TRUNC(MONTHS_BETWEEN(RSS.D8DEB, SDO.DANAIS)/12) BETWEEN 25 AND 64 THEN '25_64'
         WHEN TRUNC(MONTHS_BETWEEN(RSS.D8DEB, SDO.DANAIS)/12) >= 65 THEN '65_ET_PLUS'
         ELSE 'HORS_TRANCHE'
       END AS TRANCHE_AGE,
       COUNT(DISTINCT RSM.NOIP) AS NB_PATIENTS
FROM RSM, RUM, RSS, SDO
WHERE RUM.NORES = RSM.KYRES
  AND RSS.KYRSS = RUM.KYRSS
  AND SDO.KYNODA = RSM.NODA
  AND RSM.CDURM_P = '071S'
  AND RSS.D8DEB >= TO_DATE('01/01/2025','DD/MM/YYYY')
  AND RSS.D8DEB <  TO_DATE('01/01/2026','DD/MM/YYYY')
  AND (
       RSM.CDDNC9 LIKE 'Z448%'
       OR RSM.CDRELIE LIKE 'Z448%'
       OR EXISTS (
            SELECT 1
            FROM DGN
            WHERE DGN.KYRES = RSM.KYRES
              AND DGN.KYCDDG LIKE 'Z448%'
       )
  )
GROUP BY CASE
           WHEN TRUNC(MONTHS_BETWEEN(RSS.D8DEB, SDO.DANAIS)/12) BETWEEN 15 AND 24 THEN '15_24'
           WHEN TRUNC(MONTHS_BETWEEN(RSS.D8DEB, SDO.DANAIS)/12) BETWEEN 25 AND 64 THEN '25_64'
           WHEN TRUNC(MONTHS_BETWEEN(RSS.D8DEB, SDO.DANAIS)/12) >= 65 THEN '65_ET_PLUS'
           ELSE 'HORS_TRANCHE'
         END
ORDER BY 1;



SELECT CASE
         WHEN SUBSTR(SDO.CDRESI,1,2) = '99' THEN 'ETRANGER'
         WHEN SUBSTR(SDO.CDRESI,1,2) IN ('75','77','78','91','92','93','94','95','60')
           THEN SUBSTR(SDO.CDRESI,1,2)
         ELSE 'PROVINCE_DOM_TOM'
       END AS LIEU_RESIDENCE,
       COUNT(DISTINCT RSM.NOIP) AS NB_PATIENTS
FROM RSM, RUM, RSS, SDO
WHERE RUM.NORES = RSM.KYRES
  AND RSS.KYRSS = RUM.KYRSS
  AND SDO.KYNODA = RSM.NODA
  AND RSM.CDURM_P = '071S'
  AND RSS.D8DEB >= TO_DATE('01/01/2025','DD/MM/YYYY')
  AND RSS.D8DEB <  TO_DATE('01/01/2026','DD/MM/YYYY')
  AND (
       RSM.CDDNC9 LIKE 'Z448%'
       OR RSM.CDRELIE LIKE 'Z448%'
       OR EXISTS (
            SELECT 1
            FROM DGN
            WHERE DGN.KYRES = RSM.KYRES
              AND DGN.KYCDDG LIKE 'Z448%'
       )
  )
GROUP BY CASE
           WHEN SUBSTR(SDO.CDRESI,1,2) = '99' THEN 'ETRANGER'
           WHEN SUBSTR(SDO.CDRESI,1,2) IN ('75','77','78','91','92','93','94','95','60')
             THEN SUBSTR(SDO.CDRESI,1,2)
           ELSE 'PROVINCE_DOM_TOM'
         END
ORDER BY NB_PATIENTS DESC, LIEU_RESIDENCE;
