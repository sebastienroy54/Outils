SELECT DISTINCT
    RSM.NOIP,
    RSM.D8SOUE,
    DGN.KYCDDG
FROM RSM, DGN, SDO S
WHERE RSM.KYRES = DGN.KYRES(+)
    AND RSM.KYNODA = S.KYNODA
    AND DGN.KYCDDG IN ('O95')      -- ton filtre O95
    AND S.DANAIS < TO_DATE('01/01/2012','DD/MM/YYYY')   -- élimine les bébés
    AND RSM.D8SOUE BETWEEN TO_DATE('01/01/2025','DD/MM/YYYY') AND SYSDATE;
