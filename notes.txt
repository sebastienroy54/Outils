SELECT COUNT(DISTINCT sp.id_patient) nb_patients
FROM soins_pall sp
JOIN sejours s
  ON sp.id_sejour = s.id_sejour
WHERE s.service_referent IS NOT NULL;

SELECT COUNT(DISTINCT sp.id_patient) nb_patients
FROM soins_pall sp
WHERE sp.service_appelant IS NOT NULL;

SELECT
    COUNT(DISTINCT sp.id_patient) AS total_patients,
    COUNT(DISTINCT CASE WHEN sp.service_appelant IS NOT NULL THEN sp.id_patient END) AS avec_service_appelant,
    COUNT(DISTINCT CASE WHEN s.service_referent IS NOT NULL THEN sp.id_patient END) AS avec_service_referent
FROM soins_pall sp
LEFT JOIN sejours s
       ON sp.id_sejour = s.id_sejour;
