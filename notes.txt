from pathlib import Path

import pandas as pd
from sqlalchemy import text

from connexion_bases import get_dimbdd_engine, get_dimbdd_config


OUTPUT_DIR = Path("exports_bdd")
OUTPUT_DIR.mkdir(exist_ok=True)


def get_schema() -> str:
    """
    Récupère le schéma depuis le .credentials.
    Si absent, utilise 'public'.
    """
    cfg = get_dimbdd_config()
    return cfg.get("schema") or "public"


def get_engine():
    """
    Retourne l'engine SQLAlchemy pour dimbdd.
    """
    return get_dimbdd_engine()


def lister_tables() -> pd.DataFrame:
    """
    Liste les tables du schéma cible avec éventuelle description PostgreSQL.
    """
    engine = get_engine()
    schema = get_schema()

    query = text("""
        SELECT
            n.nspname AS schema_name,
            c.relname AS table_name,
            obj_description(c.oid, 'pg_class') AS table_description
        FROM pg_class c
        JOIN pg_namespace n
            ON n.oid = c.relnamespace
        WHERE c.relkind = 'r'
          AND n.nspname = :schema
        ORDER BY c.relname
    """)

    with engine.connect() as conn:
        df = pd.read_sql(query, conn, params={"schema": schema})

    return df


def chercher_tables(mot_cle: str) -> pd.DataFrame:
    """
    Recherche les tables contenant un mot-clé.
    """
    tables = lister_tables().copy()

    if tables.empty:
        return tables

    mask = tables["table_name"].str.contains(mot_cle, case=False, na=False)
    return tables.loc[mask].sort_values("table_name").reset_index(drop=True)


def decrire_table(table_name: str) -> pd.DataFrame:
    """
    Décrit les colonnes d'une table :
    nom, type, nullable, défaut, description éventuelle.
    """
    engine = get_engine()
    schema = get_schema()

    query = text("""
        SELECT
            cols.table_schema,
            cols.table_name,
            cols.ordinal_position,
            cols.column_name,
            cols.data_type,
            cols.udt_name,
            cols.is_nullable,
            cols.column_default,
            pgd.description AS column_description
        FROM information_schema.columns cols
        LEFT JOIN pg_catalog.pg_statio_all_tables st
            ON st.schemaname = cols.table_schema
           AND st.relname = cols.table_name
        LEFT JOIN pg_catalog.pg_description pgd
            ON pgd.objoid = st.relid
           AND pgd.objsubid = cols.ordinal_position
        WHERE cols.table_schema = :schema
          AND cols.table_name = :table_name
        ORDER BY cols.ordinal_position
    """)

    with engine.connect() as conn:
        df = pd.read_sql(
            query,
            conn,
            params={"schema": schema, "table_name": table_name},
        )

    return df


def compter_lignes(table_name: str) -> int | None:
    """
    Retourne le nombre de lignes d'une table.
    Peut être coûteux sur très grosses tables.
    """
    engine = get_engine()
    schema = get_schema()

    if schema:
        full_table_name = f'"{schema}"."{table_name}"'
    else:
        full_table_name = f'"{table_name}"'

    query = text(f"SELECT COUNT(*) AS nb FROM {full_table_name}")

    try:
        with engine.connect() as conn:
            result = conn.execute(query).scalar()
        return int(result) if result is not None else None
    except Exception as e:
        print(f"Impossible de compter les lignes pour {table_name} : {e}")
        return None


def apercu_table(table_name: str, limit: int = 20) -> pd.DataFrame:
    """
    Retourne un aperçu des premières lignes d'une table.
    """
    engine = get_engine()
    schema = get_schema()

    if schema:
        full_table_name = f'"{schema}"."{table_name}"'
    else:
        full_table_name = f'"{table_name}"'

    query = text(f"""
        SELECT *
        FROM {full_table_name}
        LIMIT :limit
    """)

    with engine.connect() as conn:
        df = pd.read_sql(query, conn, params={"limit": limit})

    return df


def resume_tables_avec_nb_lignes(limit_tables: int | None = None) -> pd.DataFrame:
    """
    Construit un résumé des tables avec nombre de lignes.
    Attention : peut être long si beaucoup de tables.
    """
    tables = lister_tables().copy()

    if limit_tables is not None:
        tables = tables.head(limit_tables).copy()

    row_counts = []
    for table_name in tables["table_name"]:
        print(f"Comptage des lignes : {table_name}")
        nb = compter_lignes(table_name)
        row_counts.append(nb)

    tables["row_count"] = row_counts
    return tables


def export_table_description_to_excel(table_name: str, include_preview: bool = True, preview_limit: int = 20) -> Path:
    """
    Exporte la description d'une table dans un Excel.
    """
    output_path = OUTPUT_DIR / f"description_{table_name}.xlsx"

    df_desc = decrire_table(table_name)
    df_preview = apercu_table(table_name, limit=preview_limit) if include_preview else pd.DataFrame()

    with pd.ExcelWriter(output_path, engine="openpyxl") as writer:
        df_desc.to_excel(writer, sheet_name="description", index=False)

        if include_preview and not df_preview.empty:
            df_preview.to_excel(writer, sheet_name="apercu", index=False)

    return output_path


def export_database_overview_to_excel(
    output_filename: str = "exploration_dimbdd.xlsx",
    include_row_counts: bool = False,
    preview_table_names: list[str] | None = None,
    preview_limit: int = 20,
) -> Path:
    """
    Exporte un fichier Excel global avec :
    - tables
    - descriptions des tables demandées
    - aperçus des tables demandées

    include_row_counts :
        True -> calcule le nombre de lignes pour toutes les tables
        False -> exporte juste la liste des tables
    """
    output_path = OUTPUT_DIR / output_filename

    if include_row_counts:
        df_tables = resume_tables_avec_nb_lignes()
    else:
        df_tables = lister_tables()

    preview_table_names = preview_table_names or []

    with pd.ExcelWriter(output_path, engine="openpyxl") as writer:
        df_tables.to_excel(writer, sheet_name="tables", index=False)

        for table_name in preview_table_names:
            print(f"Export de la table : {table_name}")

            try:
                df_desc = decrire_table(table_name)
                sheet_desc = f"desc_{table_name}"[:31]
                df_desc.to_excel(writer, sheet_name=sheet_desc, index=False)
            except Exception as e:
                print(f"Erreur description {table_name} : {e}")

            try:
                df_preview = apercu_table(table_name, limit=preview_limit)
                sheet_preview = f"data_{table_name}"[:31]
                df_preview.to_excel(writer, sheet_name=sheet_preview, index=False)
            except Exception as e:
                print(f"Erreur aperçu {table_name} : {e}")

    return output_path


def export_all_table_descriptions_to_excel(
    output_filename: str = "dictionnaire_tables_dimbdd.xlsx",
    max_tables: int | None = None,
) -> Path:
    """
    Exporte un Excel avec :
    - un onglet 'tables'
    - un onglet par table contenant la description des colonnes

    Attention :
    si tu as beaucoup de tables, le fichier peut devenir gros.
    """
    output_path = OUTPUT_DIR / output_filename
    tables = lister_tables()

    if max_tables is not None:
        tables = tables.head(max_tables).copy()

    with pd.ExcelWriter(output_path, engine="openpyxl") as writer:
        tables.to_excel(writer, sheet_name="tables", index=False)

        for table_name in tables["table_name"]:
            print(f"Description de {table_name}")
            try:
                df_desc = decrire_table(table_name)
                sheet_name = f"{table_name}"[:31]
                if df_desc.empty:
                    df_desc = pd.DataFrame({"info": [f"Aucune colonne trouvée pour {table_name}"]})
                df_desc.to_excel(writer, sheet_name=sheet_name, index=False)
            except Exception as e:
                df_error = pd.DataFrame({"erreur": [str(e)]})
                sheet_name = f"{table_name}"[:31]
                df_error.to_excel(writer, sheet_name=sheet_name, index=False)

    return output_path

    print("=== Connexion et schéma utilisés ===")

    print(f"\n=== Description de {table_test} ===")
    print(decrire_table(table_test).head(30))

    print(f"\n=== Aperçu de {table_test} ===")
    print(apercu_table(table_test, limit=5))

    print("\n=== Export Excel simple ===")
    path1 = export_database_overview_to_excel(
        output_filename="exploration_dimbdd.xlsx",

        preview_limit=20,
    )
    print(f"Fichier créé : {path1.resolve()}")

    print("\n=== Export dictionnaire des tables ===")
    path2 = export_all_table_descriptions_to_excel(
        output_filename="dictionnaire_tables_dimbdd.xlsx",
        max_tables=30,  # retire ce paramètre ou mets None pour tout exporter
    )
    print(f"Fichier créé : {path2.resolve()}")


if __name__ == "__main__":
    main()
