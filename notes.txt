SELECT
    sp.service_appelant,
    COUNT(DISTINCT sp.id_patient) AS nb_patients,
    COUNT(DISTINCT sp.id_soins_pall) AS nb_interventions
FROM soins_pall sp
WHERE sp.date_intervention BETWEEN '2016-01-01' AND CURRENT_DATE
  AND sp.service_appelant IS NOT NULL
GROUP BY sp.service_appelant
ORDER BY sp.service_appelant;


SELECT
    s.service_referent,
    COUNT(DISTINCT sp.id_patient) AS nb_patients,
    COUNT(DISTINCT sp.id_soins_pall) AS nb_interventions
FROM soins_pall sp
JOIN sejours s
    ON sp.id_sejour = s.id_sejour
WHERE sp.date_intervention BETWEEN '2016-01-01' AND CURRENT_DATE
  AND s.service_referent IS NOT NULL
GROUP BY s.service_referent
ORDER BY s.service_referent;
