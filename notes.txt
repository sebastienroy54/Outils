from __future__ import annotations

import csv
import re
import sys
import unicodedata
from pathlib import Path
from typing import Dict, Iterable, List, Optional, Tuple

try:
    from openpyxl import load_workbook
except ImportError as exc:
    raise SystemExit("Le module openpyxl est requis pour lire le fichier .xlsm.") from exc


XLSM_FILENAME = "CCAM_V82_-02.01.2026._VF(1).xlsm"
CSV_FILENAME = "sejours_actes_j0_uma_2024_2025.csv"
OUTPUT_DETAIL = "sejours_actes_j0_uma_2024_2025_adc_detail.csv"
OUTPUT_SUMMARY = "sejours_actes_j0_uma_2024_2025_adc_resume.csv"
TARGET_REGROUPEMENT = "ADC"


def normalize_text(value: object) -> str:
    """Normalise un texte pour comparer des en-têtes de façon robuste."""
    if value is None:
        return ""
    text = str(value)
    text = text.replace("\ufeff", " ").replace("\xa0", " ")
    text = "".join(ch for ch in text if ch.isprintable())
    text = unicodedata.normalize("NFKD", text)
    text = "".join(ch for ch in text if not unicodedata.combining(ch))
    text = text.strip().lower()
    text = re.sub(r"\s+", " ", text)
    return text


def canonical_code(value: object) -> str:
    """Normalise un code CCAM pour comparaison."""
    if value is None:
        return ""
    text = str(value).strip().upper()
    text = text.replace("\ufeff", "").replace("\xa0", "")
    text = re.sub(r"\s+", "", text)
    return text


def detect_header_indexes(header_values: List[object]) -> Tuple[int, int]:
    normalized = [normalize_text(v) for v in header_values]

    code_candidates = []
    regroupement_candidates = []

    for idx, name in enumerate(normalized):
        compact = name.replace(" ", "")
        if compact == "code" or compact.startswith("code"):
            code_candidates.append(idx)
        if compact == "regroupement" or compact.startswith("regroupement"):
            regroupement_candidates.append(idx)

    if not code_candidates:
        for idx, name in enumerate(normalized):
            if "code" in name:
                code_candidates.append(idx)
    if not regroupement_candidates:
        for idx, name in enumerate(normalized):
            if "regroupement" in name:
                regroupement_candidates.append(idx)

    if not code_candidates or not regroupement_candidates:
        raise ValueError(
            "Impossible de détecter les colonnes 'Code' et/ou 'Regroupement' dans le .xlsm."
        )

    return code_candidates[0], regroupement_candidates[0]


def find_code_regroupement_sheet(xlsm_path: Path) -> Tuple[str, int, int]:
    wb = load_workbook(xlsm_path, read_only=True, data_only=True, keep_vba=True)
    try:
        for sheet_name in wb.sheetnames:
            ws = wb[sheet_name]
            max_scan_rows = min(ws.max_row or 0, 30)
            for row_idx, row in enumerate(
                ws.iter_rows(min_row=1, max_row=max_scan_rows, values_only=True), start=1
            ):
                row_values = list(row)
                try:
                    code_idx, reg_idx = detect_header_indexes(row_values)
                    return sheet_name, row_idx, code_idx
                except ValueError:
                    continue
    finally:
        wb.close()

    raise ValueError(
        "Aucune feuille ne contient un en-tête détectable avec 'Code' et 'Regroupement'."
    )


def load_adc_codes(xlsm_path: Path) -> Dict[str, str]:
    sheet_name, header_row, code_idx = find_code_regroupement_sheet(xlsm_path)
    wb = load_workbook(xlsm_path, read_only=True, data_only=True, keep_vba=True)
    try:
        ws = wb[sheet_name]
        header = [cell for cell in next(ws.iter_rows(min_row=header_row, max_row=header_row, values_only=True))]
        code_idx, reg_idx = detect_header_indexes(header)

        adc_codes: Dict[str, str] = {}
        for row in ws.iter_rows(min_row=header_row + 1, values_only=True):
            if row is None:
                continue
            code = canonical_code(row[code_idx] if code_idx < len(row) else None)
            regroupement = normalize_text(row[reg_idx] if reg_idx < len(row) else None)
            if not code:
                continue
            if regroupement == TARGET_REGROUPEMENT.lower():
                adc_codes[code] = TARGET_REGROUPEMENT
        return adc_codes
    finally:
        wb.close()


def read_sql_csv(csv_path: Path) -> List[dict]:
    rows: List[dict] = []
    with csv_path.open("r", encoding="utf-8-sig", newline="") as f:
        reader = csv.DictReader(f, delimiter=";")
        expected = {"ANNEE", "KYRES", "KYRSS", "UMA", "DATE_ENTREE_RSS", "CODE_CCAM", "DATE_REAL_ACTE"}
        missing = expected.difference(reader.fieldnames or [])
        if missing:
            raise ValueError(
                "Colonnes attendues absentes du CSV SQL : " + ", ".join(sorted(missing))
            )
        for row in reader:
            row = {k: (v.strip() if isinstance(v, str) else v) for k, v in row.items()}
            row["CODE_CCAM_NORM"] = canonical_code(row.get("CODE_CCAM", ""))
            rows.append(row)
    return rows


def write_csv(path: Path, rows: Iterable[dict], fieldnames: List[str]) -> None:
    with path.open("w", encoding="utf-8-sig", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames, delimiter=";")
        writer.writeheader()
        for row in rows:
            writer.writerow(row)


def main() -> int:
    base_dir = Path(__file__).resolve().parent
    xlsm_path = base_dir / XLSM_FILENAME
    csv_path = base_dir / CSV_FILENAME
    detail_path = base_dir / OUTPUT_DETAIL
    summary_path = base_dir / OUTPUT_SUMMARY

    if not xlsm_path.exists():
        raise FileNotFoundError(f"Fichier .xlsm introuvable : {xlsm_path}")
    if not csv_path.exists():
        raise FileNotFoundError(f"Fichier CSV introuvable : {csv_path}")

    adc_codes = load_adc_codes(xlsm_path)
    if not adc_codes:
        raise ValueError("Aucun code trouvé avec le regroupement ADC dans le fichier .xlsm.")

    sql_rows = read_sql_csv(csv_path)

    matched_rows: List[dict] = []
    unmatched_codes = set()
    all_sql_codes = set()

    for row in sql_rows:
        code_norm = row["CODE_CCAM_NORM"]
        if code_norm:
            all_sql_codes.add(code_norm)
        if code_norm in adc_codes:
            new_row = dict(row)
            new_row["REGROUPEMENT"] = adc_codes[code_norm]
            matched_rows.append(new_row)
        else:
            if code_norm:
                unmatched_codes.add(code_norm)

    detail_fieldnames = [
        "ANNEE",
        "KYRES",
        "KYRSS",
        "UMA",
        "DATE_ENTREE_RSS",
        "CODE_CCAM",
        "DATE_REAL_ACTE",
        "REGROUPEMENT",
    ]
    write_csv(detail_path, matched_rows, detail_fieldnames)

    summary_rows: List[dict] = []
    for year in sorted({row["ANNEE"] for row in sql_rows}):
        year_rows = [row for row in matched_rows if row["ANNEE"] == year]
        distinct_sejours = {row["KYRES"] for row in year_rows if row.get("KYRES")}
        distinct_rss = {row["KYRSS"] for row in year_rows if row.get("KYRSS")}
        distinct_codes = {row["CODE_CCAM_NORM"] for row in year_rows if row.get("CODE_CCAM_NORM")}
        summary_rows.append(
            {
                "ANNEE": year,
                "NB_LIGNES_ACTES_ADC_J0": len(year_rows),
                "NB_SEJOURS_DISTINCTS_ADC_J0": len(distinct_sejours),
                "NB_RSS_DISTINCTS_ADC_J0": len(distinct_rss),
                "NB_CODES_CCAM_ADC_DIFFERENTS": len(distinct_codes),
            }
        )

    write_csv(
        summary_path,
        summary_rows,
        [
            "ANNEE",
            "NB_LIGNES_ACTES_ADC_J0",
            "NB_SEJOURS_DISTINCTS_ADC_J0",
            "NB_RSS_DISTINCTS_ADC_J0",
            "NB_CODES_CCAM_ADC_DIFFERENTS",
        ],
    )

    total_distinct_sejours = {row["KYRES"] for row in matched_rows if row.get("KYRES")}

    print("Traitement terminé.")
    print(f"Fichier xlsm lu : {xlsm_path.name}")
    print(f"Fichier CSV lu : {csv_path.name}")
    print(f"Codes ADC trouvés dans le référentiel : {len(adc_codes)}")
    print(f"Lignes SQL lues : {len(sql_rows)}")
    print(f"Lignes ADC retenues : {len(matched_rows)}")
    print(f"Séjours distincts ADC J0 (toutes années) : {len(total_distinct_sejours)}")
    print(f"Codes SQL non retrouvés en ADC : {len(unmatched_codes)}")
    print(f"Détail exporté : {detail_path.name}")
    print(f"Résumé exporté : {summary_path.name}")

    if unmatched_codes:
        unmatched_path = base_dir / "codes_ccam_non_retrouves_adc.txt"
        with unmatched_path.open("w", encoding="utf-8") as f:
            for code in sorted(unmatched_codes):
                f.write(code + "\n")
        print(f"Liste des codes non retrouvés exportée : {unmatched_path.name}")

    return 0


if __name__ == "__main__":
    try:
        raise SystemExit(main())
    except Exception as exc:  # noqa: BLE001
        print(f"ERREUR : {exc}", file=sys.stderr)
        raise
