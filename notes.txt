SELECT
    COUNT(DISTINCT H.NOIP) AS NB_IPP
FROM
    RSM R,
    EPI E,
    HJO H
WHERE
    R.KYRES = E.NORES
    AND E.NOHJO = H.NOHJO
    AND R.CDURM_P = '520J'
    AND R.CDDNC9 IN ('M545','M5450','M5456','M5457','M5458','M5459')
    AND H.DAEXEC >= TO_DATE('01/01/2024','DD/MM/YYYY')
    AND H.DAEXEC <  TO_DATE('01/01/2025','DD/MM/YYYY');
