def run_groupage(
    rss_input_path: Path,
    year: int,
    typ_etab: int = 1
) -> dict:
    year_dirs = get_year_dirs(year)
    grpmco_exe = find_grpmco_exe()

    mcofg_path = write_mcofg_ini(
        rss_input_path=rss_input_path,
        year=year,
        typ_etab=typ_etab
    )

    result = subprocess.run(
        [str(grpmco_exe), str(mcofg_path)],
        cwd=str(grpmco_exe.parent),
        capture_output=True,
        text=True,
        shell=False
    )

    stdout_log = year_dirs["logs"] / f"{rss_input_path.stem}_stdout.txt"
    stderr_log = year_dirs["logs"] / f"{rss_input_path.stem}_stderr.txt"

    stdout_log.write_text(result.stdout or "", encoding="utf-8", errors="ignore")
    stderr_log.write_text(result.stderr or "", encoding="utf-8", errors="ignore")

    expected_outputs = {
        "grp": year_dirs["output"] / f"{rss_input_path.name}.grp",
        "err": year_dirs["output"] / f"{rss_input_path.name}.err",
        "log": year_dirs["output"] / f"{rss_input_path.name}.log",
        "sup": year_dirs["output"] / f"{rss_input_path.name}.sup",
        "rds": year_dirs["output"] / f"{rss_input_path.name}.rds",
        "stp": year_dirs["output"] / f"{rss_input_path.name}.stp.txt",
    }

    existing_outputs = {
        name: path
        for name, path in expected_outputs.items()
        if path.exists()
    }

    mcofg_results = read_mcofg_results(mcofg_path)

    return {
        "returncode": result.returncode,
        "stdout": result.stdout,
        "stderr": result.stderr,
        "grpmco_exe": grpmco_exe,
        "mcofg_path": mcofg_path,
        "mcofg_results": mcofg_results,
        "stdout_log": stdout_log,
        "stderr_log": stderr_log,
        "outputs": existing_outputs,
    }
