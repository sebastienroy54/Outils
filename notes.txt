SELECT COUNT(*)
FROM EPI
WHERE EPI.CHPPMSI = 'SSR'
AND EPI.D8SOUE >= TO_DATE('01/03/2026','DD/MM/YYYY')
AND EPI.D8SOUE < TO_DATE('01/04/2026','DD/MM/YYYY')
AND EPI.MDEEUE IS NOT NULL
AND EPI.MDSOUE IS NOT NULL;


SELECT EPI.MDEEUE, EPI.MDSOUE, COUNT(*)
FROM EPI
WHERE EPI.CHPPMSI = 'SSR'
AND EPI.D8SOUE >= TO_DATE('01/03/2026','DD/MM/YYYY')
AND EPI.D8SOUE < TO_DATE('01/04/2026','DD/MM/YYYY')
GROUP BY EPI.MDEEUE, EPI.MDSOUE
ORDER BY COUNT(*) DESC;
