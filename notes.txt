import pandas as pd
import streamlit as st
from io import BytesIO
import zipfile
import re

def main():
    st.title("Croisement RAA")

    # Récupération du fichier
    agenda_file = st.file_uploader("Téléversez votre fichier Excel ici contenant les noms des praticiens", type=["xlsx"])
    ipp_file = st.file_uploader("Téléversez votre fichier Excel ici contenant des manquants", type=["xlsx"])

    if agenda_file and ipp_file:
        try:
            # Création d'un nouveau dataframe pour y mettre les colonnes avec les données voulues
            try:
                df_raw = pd.read_excel(ipp_file, header=None)
                colonnes = ['IPP', 'Date', 'Nom', 'Prénom', 'Unité', 'EDGAR']

                if df_raw.shape[1] != len(colonnes):
                    raise ValueError(
                        f"Le fichier IPP contient {df_raw.shape[1]} colonnes "
                        f"mais {len(colonnes)} sont attendues ({colonnes})."
                    )
                ipp_df = pd.DataFrame(df_raw.values, columns=colonnes)
            except Exception as e:
                st.error(f"Erreur lors de la lecture du fichier IPP : {e}")
                return
            
            try:
                agenda = pd.read_excel(agenda_file)
            except Exception as e:
                st.error(f"Erreur lors de la lecture du fichier Agenda : {e}")
                return

            # Formatage des données
            try:
                for col in ['Patient', 'Acte EDGAR', 'IPP', 'EDGAR']:
                    if col in agenda.columns:
                        agenda[col] = agenda[col].astype(str).str.strip()
                    if col in ipp_df.columns:
                        ipp_df[col] = ipp_df[col].astype(str).str.strip()
            
                if 'Date RDV' in agenda.columns:
                    agenda['Date RDV'] = pd.to_datetime(agenda['Date RDV'], errors='coerce')
                if 'Date' in ipp_df:
                    ipp_df['Date'] = pd.to_datetime(ipp_df['Date'], errors='coerce')
            except Exception as e:
                st.error(f"Erreur lors du formatage des données : {e}")
                return

            # Paramètres pour la fusion entre IPP et Patient
            try:
                merged = pd.merge(
                    ipp_df,
                    agenda,
                    left_on='IPP',
                    right_on='Patient',
                    how='inner',
                    suffixes=('_ipp', '_agenda')
                )
            except Exception as e:
                st.error(f"Erreur lors de la fusion des fichiers : {e}")
                return

            # Condition sur les dates et les EDGAR
            try:
                condition = (
                    (merged["Date RDV"] == merged['Date']) &
                    (merged["Acte EDGAR"] == merged['EDGAR'])
                )

                resultat = merged.loc[condition, ['IPP', 'Nom', 'Prénom', 'NDA', 'Date', 'Agenda', 'EDGAR']].copy()
                resultat['Date'] = pd.to_datetime(resultat['Date'], errors='coerce').dt.strftime('%d/%m/%Y')
            except KeyError as e:
                st.error(f"Colonne manquante pour la condition : {e}")
                return
            except Exception as e:
                st.error(f"Erreur lors de l'application des conditions : {e}")
                return

            try:
                if not resultat.empty:
                    # Extraction du nom praticien : tout ce qu'il y a après le "-"
                    resultat["Praticien"] = resultat["Agenda"].astype(str).str.split("-", n=1).str[1].str.strip()

                    # Sécurisation du nom de fichier
                    def nettoyer_nom_fichier(nom):
                        nom = str(nom).strip()
                        nom = re.sub(r'[\\/*?:"<>|]', "_", nom)
                        nom = re.sub(r"\s+", "_", nom)
                        return nom

                    # Création du ZIP en mémoire
                    zip_buffer = BytesIO()

                    with zipfile.ZipFile(zip_buffer, "w", zipfile.ZIP_DEFLATED) as zip_file:
                        for praticien, df_praticien in resultat.groupby("Praticien"):
                            nom_fichier = f"{nettoyer_nom_fichier(praticien)}_manquant_RAA.xlsx"

                            excel_buffer = BytesIO()
                            with pd.ExcelWriter(excel_buffer, engine='openpyxl') as writer:
                                df_export = df_praticien.drop(columns=["Praticien"])
                                df_export.to_excel(writer, index=False, sheet_name='Résultats_manquants_60R')
                            excel_buffer.seek(0)

                            zip_file.writestr(nom_fichier, excel_buffer.getvalue())

                    zip_buffer.seek(0)

                    st.download_button(
                        label="Télécharger le ZIP des fichiers par praticien",
                        data=zip_buffer,
                        file_name="Manquants_RAA_par_praticien.zip",
                        mime="application/zip"
                    )
                else:
                    st.info("Aucune correspondance trouvée selon les critères définis.")
            except Exception as e:
                st.error(f"Erreur lors de la génération du ZIP : {e}")
                return

        except Exception as e:
            st.error(f"Erreur inattendue : {type(e).__name__} - {e}")

if __name__ == "__main__":
    main()
