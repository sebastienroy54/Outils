SELECT COUNT(*) NB
FROM EPI
WHERE EPI.UE = '021X762';

SELECT COUNT(*) NB
FROM RSM
WHERE RSM.UE = '021X762';

SELECT OWNER, TABLE_NAME, COLUMN_NAME
FROM ALL_TAB_COLUMNS
WHERE (TABLE_NAME IN ('EPI','RSM') AND (UPPER(COLUMN_NAME) LIKE '%UH%' OR UPPER(COLUMN_NAME) LIKE '%UE%'))
ORDER BY OWNER, TABLE_NAME, COLUMN_NAME;
