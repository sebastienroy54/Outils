def clean_digits_id(x) -> str:
    """
    Normalise un identifiant (IPP/NDA/NIP/NAS) :
    - strip, supprime .0
    - supprime H si H61...
    - garde uniquement les chiffres
    """
    s = "" if x is None else str(x).strip()
    if not s:
        return ""
    s = s.replace("\u00A0", " ").strip()

    s = s.upper()
    s = re.sub(r"^H(?=61\d+)", "", s)  # H615... -> 615...
    s = re.sub(r"\D+", "", s)         # digits only
    return s
