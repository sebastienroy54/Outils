def find_grpmco_exe() -> Path:
    """
    Cherche grpmco.exe sans dépendre d'un chemin absolu unique.

    À long terme :
    - privilégier stage/pdg_mco/fg/grpmco.exe

    Actuellement :
    """

    candidates = [
        BASE_DIR / "pdg_mco" / "fg" / "grpmco.exe",
    ]

    for candidate in candidates:
        if candidate.exists():
            return candidate

    searched = "\n".join(str(c) for c in candidates)
    raise FileNotFoundError(
        "grpmco.exe introuvable. Chemins testés :\n" + searched
    )
