SELECT *
FROM (
    SELECT
        ACS.CDPHCO,
        MAX(ACS.DACO) AS DERNIERE_CONSULTATION
    FROM ACS
    WHERE ACS.NOUFCO = '670'
      AND ACS.TYMAJ <> 'D'
      AND ACS.CDPHCO IS NOT NULL
    GROUP BY ACS.CDPHCO
    ORDER BY MAX(ACS.DACO) DESC
)
WHERE ROWNUM <= 10
;


SELECT *
FROM (
    SELECT
        ACS.CDPHCO,
        COUNT(DISTINCT ACS.NOACS) AS NB_CONSULTATIONS,
        MAX(ACS.DACO) AS DERNIERE_CONSULTATION
    FROM ACS
    WHERE ACS.NOUFCO = '670'
      AND ACS.TYMAJ <> 'D'
      AND ACS.CDPHCO IS NOT NULL
    GROUP BY ACS.CDPHCO
    ORDER BY MAX(ACS.DACO) DESC
)
WHERE ROWNUM <= 10
;
