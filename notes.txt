SELECT RSM.NOIP        AS IPP,
       SDO.NMMAL       AS NOM,
       SDO.NMPMAL      AS PRENOM,
       SDO.DANAIS      AS DDN,
       RSM.NODA        AS NDA,
       RSM.D8EEUE      AS DATE_ENTREE,
       RSM.D8SOUE      AS DATE_SORTIE,
       RAC.KYCDAC      AS ACTE_CCAM,
       RAC.D8REAL      AS DATE_REAL_ACTE
FROM RSM, RAC, SDO
WHERE RAC.KYRES = RSM.KYRES
  AND SDO.KYNODA = RSM.NODA
  AND RSM.CDURM_P IN ('150','150J','151J')
  AND RAC.KYCDAC = 'AMQP009'
  AND RAC.D8REAL >= TO_DATE('01/07/2025','DD/MM/YYYY')
  AND RAC.D8REAL < TRUNC(SYSDATE) + 1
ORDER BY RSM.NOIP, RSM.NODA, RAC.D8REAL;
