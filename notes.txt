SELECT DISTINCT
    RSM.NODA AS NDA,
    RSM.NOIP AS IPP,
    EPI.TYEPI AS TYPE,
    RSM.MDENTR AS MODE_ENTREE,
    RSM.MDSOR AS MODE_SORTIE,
    RSM.D8EEUE AS DATE_ENTREE,
    RSM.D8SOUE AS DATE_SORTIE
FROM
    RSM,
    EPI
WHERE
    RSM.KYRES = EPI.NORES
    AND EPI.CHPPMSI = 'SSR'
    AND RSM.D8SOUE >= TO_DATE('01/01/2025','DD/MM/YYYY')
    AND RSM.D8SOUE <  TO_DATE('01/01/2026','DD/MM/YYYY')
    AND ROWNUM <= 20
ORDER BY
    RSM.D8SOUE;
