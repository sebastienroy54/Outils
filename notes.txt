import pandas as pd

# Liste des codes autorisés
codes_autorises = {
    "HEQE002",
    "HHFE002",
    "HHQE002",
    "HHQE005",
    "ZZLP025",
    "HJQE001",
    "HHFE004",
    "EHNE001",
    "HHFE001",
    "HEFE002",
    "HESE002",
    "HHQE004",
    "HGSE003",
    "HHFE006",
    "HEFE001",
    "HHSE002"
}

# Lire le fichier Excel
df = pd.read_excel("export_infomed.xlsx")

# Fonction qui vérifie si un code autorisé est présent
def contient_code(valeur):
    if pd.isna(valeur):
        return False
    
    # Séparer les actes (souvent séparés par ;)
    actes = str(valeur).split(";")
    
    for acte in actes:
        # enlever le (A) éventuel
        code = acte.replace("(A)", "").strip()
        if code in codes_autorises:
            return True
    return False

# Filtrer les lignes
df_filtre = df[df["Acte codant"].apply(contient_code)]

# Sauvegarder le résultat
df_filtre.to_excel("export_infomed_filtre.xlsx", index=False)

print("Filtrage terminé. Fichier créé : export_infomed_filtre.xlsx")
