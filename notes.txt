SPOOL C:/Users/4251352/projets/resultats_requetes/simpa_activite1_2025.csv
SELECT 'IPP;ACTE;DATE_EXE;NODA;CDGHM' FROM DUAL;
SELECT DISTINCT
    SDO.KYNOIP,
    RAC.KYCDAC,
    TO_CHAR(TRUNC(RAC.D8REAL),'DD/MM/YYYY'),
    RSM.NODA,
    RSS.CDGHM
FROM
    RSM, RSS, RUM, RAC, SDO
WHERE
    RSS.KYRSS = RUM.KYRSS
    AND RUM.NORES = RSM.KYRES
    AND RSM.KYRES = RAC.KYRES
    AND RSS.NODA = SDO.KYNODA
    AND RAC.D8REAL >= TO_DATE('01/01/2025','DD/MM/YYYY')
    AND RAC.D8REAL <  TO_DATE('01/01/2026','DD/MM/YYYY')
    AND RAC.KYACTIVITE = '1'
