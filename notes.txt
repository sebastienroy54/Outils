def format_date_effet_ddmmyyyy(x) -> str:
    if x is None or (isinstance(x, float) and pd.isna(x)) or (isinstance(x, str) and x.strip() == ""):
        return "01011900"  # fallback neutre (à ajuster si tu préfères bloquer)
    dt = pd.to_datetime(x, dayfirst=True, errors="coerce")
    if pd.isna(dt):
        # tente une lecture brute si c'est déjà du type "01012005"
        s = str(x).strip()
        if re.fullmatch(r"\d{8}", s):
            return s
        return "01011900"
    return dt.strftime("%d%m%Y")


def format_nb_lits_3(x) -> str:
    if x is None or (isinstance(x, float) and pd.isna(x)) or (isinstance(x, str) and x.strip() == ""):
        return "000"
    s = str(x).strip().replace(" ", "")
    s = s.replace(",", "")
    # parfois excel peut donner "14.0"
    if re.fullmatch(r"\d+(\.0)?", s):
        s = s.split(".")[0]
    s = re.sub(r"\D", "", s)
    if s == "":
        return "000"
    return s.zfill(3)[-3:]


def build_ficum_lines_from_excel(um_df: pd.DataFrame, site5: str, finess: str) -> bytes:
    """
    Construit ficum{site}.txt à partir de l'excel.
    Colonnes attendues (ligne 3 dans excel) :
      CODE_HOPITAL, URM_SITE, URM_APHP, TYPE_AUTORISATION, DATE_EFFET, NB_LITS, TYPE_HOSPITALISATION
    """
    required = ["CODE_HOPITAL", "URM_SITE", "TYPE_AUTORISATION", "DATE_EFFET", "NB_LITS", "TYPE_HOSPITALISATION"]
    missing = [c for c in required if c not in um_df.columns]
    if missing:
        raise ValueError(f"Impossible de générer FICUM: colonnes manquantes dans l'excel: {missing}")

    df = um_df.copy()

    # garder uniquement le site concerné si l'excel contient plusieurs sites (normalement non)
    df["CODE_HOPITAL"] = df["CODE_HOPITAL"].astype(str).str.strip()
    df = df[df["CODE_HOPITAL"] == site5]

    # normalisations
    df["URM_SITE"] = df["URM_SITE"].astype(str).str.strip()
    df["TYPE_AUTORISATION"] = df["TYPE_AUTORISATION"].astype(str).str.strip().str.replace(" ", "", regex=False)
    df["TYPE_HOSPITALISATION"] = df["TYPE_HOSPITALISATION"].astype(str).str.strip().str[:1]

    # UM sur 4 chars (padding à droite)
    def um4(u: str) -> str:
        u = (u or "").strip()
        return (u + "    ")[:4]

    df["UM4"] = df["URM_SITE"].apply(um4)

    # règle spéciale CCH: remplacer "$21" par "$$$" dans l'UM (ex "$213" -> "$$$3")
    # D'après ton besoin: uniquement Cochin (00021)
    if site5 == "00021":
        df["UM4"] = df["UM4"].apply(lambda s: (s.replace("$21", "$$$", 1) if s.startswith("$21") else s))

    df["DATE8"] = df["DATE_EFFET"].apply(format_date_effet_ddmmyyyy)
    df["LITS3"] = df["NB_LITS"].apply(format_nb_lits_3)

    # construit les lignes
    lines = []
    for _, r in df.iterrows():
        line = (
            r["UM4"]
            + finess
            + r["TYPE_AUTORISATION"]
            + " "
            + r["DATE8"]
            + r["LITS3"]
            + r["TYPE_HOSPITALISATION"]
        )
        lines.append(line)

    # IMPORTANT: encodage latin-1 comme le reste, + newline final
    txt = "\n".join(lines) + ("\n" if lines else "")
    return safe_encode_singlebyte(txt)
