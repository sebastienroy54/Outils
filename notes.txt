SELECT DISTINCT
    SDO.KYNOIP AS IPP,
    RAC.KYCDAC AS ACTE,
    TO_CHAR(TRUNC(RAC.D8REAL),'DD/MM/YYYY') AS DATE_EXE,
    RSM.NODA,
    RSS.CDGHM
FROM
    RSM, RSS, RUM, RAC, SDO
WHERE
    RSS.KYRSS = RUM.KYRSS
    AND RUM.NORES = RSM.KYRES
    AND RSM.KYRES = RAC.KYRES
    AND RSS.NODA = SDO.KYNODA
    AND RAC.D8REAL >= TO_DATE('01/01/2025','DD/MM/YYYY')
    AND RAC.D8REAL <  TO_DATE('01/01/2026','DD/MM/YYYY')
    AND RAC.KYACTIVITE = '1'
    AND RSS.CDGHM LIKE '%M%'
;
