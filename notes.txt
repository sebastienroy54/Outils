import pandas as pd

# Chemin du fichier
fichier = r"C:/Users/4251352/Documents/Resultats_requetes/mucoviscidose_2025.csv"

# Lire le CSV
df = pd.read_csv(fichier)

# Compter les occurrences de chaque IPP
compte_ipp = df['IPP'].value_counts()

# Calcul des doublons (occurrences - 1)
doublons = (compte_ipp - 1).clip(lower=0).sum()

print("Nombre total de doublons IPP :", doublons)
