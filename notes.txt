from pathlib import Path
from sqlalchemy import create_engine, text


def read_credentials(path):
    creds = {}
    current_section = None

    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()

            if not line or line.startswith("#"):
                continue

            # nouvelle section [dbname]
            if line.startswith("[") and line.endswith("]"):
                current_section = line[1:-1]
                creds[current_section] = {}
                continue

            if ":" in line and current_section:
                key, value = line.split(":", 1)
                creds[current_section][key.strip()] = value.strip()

    return creds


def test_connection(name, cfg):
    try:
        user = cfg["user"]
        password = cfg["password"]
        host = cfg["host"]
        port = cfg.get("port", "5432")
        dbname = cfg["dbname"]

        url = f"postgresql+psycopg2://{user}:{password}@{host}:{port}/{dbname}"

        engine = create_engine(url)

        with engine.connect() as conn:
            conn.execute(text("SELECT 1"))

        print(f"✅ {name} : connexion OK")

    except Exception as e:
        print(f"❌ {name} : échec connexion")
        print("   ", e)


def main():

    cred_path = Path(r"C:\Users\4251352\Documents\.credentials")

    creds = read_credentials(cred_path)

    print(f"{len(creds)} bases trouvées\n")

    for name, cfg in creds.items():
        test_connection(name, cfg)


if __name__ == "__main__":
    main()
