library(stringr)
library(readr)

# Lire le fichier
file_path <- "SSR22_Anohosp.fic"
lines <- read_lines(file_path)

# Fonction pour extraire le format
extract_format <- function(line) {
  fmt <- substr(line, 17, 19)        # caractères 17 à 19
  paste0("V", str_trim(fmt))         # ajoute "V" et supprime les espaces
}

# Appliquer à toutes les lignes
formats <- sapply(lines, extract_format)

# Afficher la répartition
table(formats)
