SELECT DISTINCT
    R.KYRES,
    R.NODA,
    R.CDURM_P,
    R.D8EEUE,
    R.D8SOUE,
    R.D8RES
FROM
    RSM R
WHERE
    R.D8SOUE >= TO_DATE('01/12/2025','DD/MM/YYYY')
    AND R.D8SOUE <  TO_DATE('01/01/2026','DD/MM/YYYY')
    AND NOT EXISTS (
        SELECT 1
        FROM RUM U
        WHERE U.NORES = R.KYRES
    );
