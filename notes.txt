SELECT
    COUNT(*) AS NB_RUM
FROM
    RUM M,
    RSM R
WHERE
    M.NORES = R.KYRES
    AND R.CDURM_P = '520S'
    AND R.CDDNC9 IN (
        'M545',
        'M5450',
        'M5456',
        'M5457',
        'M5458',
        'M5459'
    )
    AND M.D8EEUE >= TO_DATE('01/01/2024','DD/MM/YYYY')
    AND M.D8EEUE <  TO_DATE('01/01/2025','DD/MM/YYYY');
