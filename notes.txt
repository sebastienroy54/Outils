SELECT CAT_INTERVENTION, COUNT(DISTINCT KYRES)
FROM RSM
WHERE CDURM_P IN ('140J','142J')
AND D8EEUE >= TO_DATE('01/01/2024','DD/MM/YYYY')
AND D8EEUE <  TO_DATE('01/01/2025','DD/MM/YYYY')
AND CDDNC9 LIKE 'O04%'
GROUP BY CAT_INTERVENTION;

SELECT CODEOR, COUNT(DISTINCT KYRES)
FROM RSM
WHERE CDURM_P IN ('140J','142J')
AND D8EEUE >= TO_DATE('01/01/2024','DD/MM/YYYY')
AND D8EEUE <  TO_DATE('01/01/2025','DD/MM/YYYY')
AND CDDNC9 LIKE 'O04%'
GROUP BY CODEOR;
