SELECT
    E.KYNODA AS NDA,
    E.NORES  AS KYRES,
    E.D8SOUE AS SORTIE_UE,
    COUNT(D.KYCDDG) AS NB_DAS
FROM
    EPI E,
    RSM R,
    DGN D
WHERE
    E.NORES = R.KYRES
    AND D.KYRES(+) = R.KYRES
    AND E.CDURM_P = '146J'
    AND E.D8SOUE >= TO_DATE('01/11/2024','DD/MM/YYYY')
    AND E.D8SOUE <  TO_DATE('01/01/2026','DD/MM/YYYY')
GROUP BY
    E.KYNODA, E.NORES, E.D8SOUE
HAVING
    COUNT(D.KYCDDG) > 1
ORDER BY
    NB_DAS DESC, SORTIE_UE;
