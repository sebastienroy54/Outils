SET LINESIZE 32767
SET HEADING ON
SET COLSEP ";"
SET PAGESIZE 0
SET FEEDBACK OFF
SET TRIMSPOOL ON
SET WRAP OFF
SET TERMOUT OFF
SPOOL C:/Users/4251352/Portail_outils/data/BHRe.csv
SET @IPP = [liste de nos IPP];
SELECT DISTINCT S.KYNOIP AS IPP, R.NODA AS NDA
FROM SDO S, RSS R
WHERE S.KYNOIP IN (@IPP)
AND S.KYNODA = R.KYNODA
AND R.D8DEB >= TO_DATE('01/01/2025','DD/MM/YYYY')
AND R.D8DEB <= TRUN(SYSDATE)
AND R.DUSEJ >= 4
AND (R.NODA('%2')
OR R.NODA('%1'))
ORDER BY NDA;
