SELECT DISTINCT R.NOIP AS IPP, R.NODA AS NDA, R.CDDNC9 AS DP, R.CDRELIE AS DR, D.KYCDDG AS DAS
FROM RSM R, DGN D
WHERE R.KYRES = D.KYRES(+)
  AND (
        R.CDDNC9 = 'O95'      -- DP
     OR R.CDRELIE = 'O95'     -- DR
     OR D.KYCDDG = 'O95'      -- DAS
  )
  AND R.D8EEUE BETWEEN TO_DATE('01/01/2013','DD/MM/YYYY') AND SYSDATE;
SPOOL OFF
