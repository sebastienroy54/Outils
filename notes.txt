SELECT RAC.KYCDAC,
       COUNT(*) AS NB_ACTES
FROM RSM, RAC
WHERE RAC.KYRES = RSM.KYRES
  AND RSM.UE = '273'
  AND RAC.KYCDAC IN ('AHQB032', 'AHQB033')
  AND RAC.D8REAL >= TO_DATE('01/01/2025','DD/MM/YYYY')
  AND RAC.D8REAL <  TO_DATE('01/06/2025','DD/MM/YYYY')
GROUP BY RAC.KYCDAC
ORDER BY RAC.KYCDAC;
