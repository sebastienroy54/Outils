SELECT R.CDURM_P AS UMA,
       COUNT(DISTINCT S.KYRSS) AS NB_RSS
FROM RSM R, RUM U, RSS S
WHERE U.NORES = R.KYRES
AND U.KYRSS = S.KYRSS
AND R.CDURM_P IN ('021','310','101','104','393','451')
AND S.D8DEB >= TO_DATE('01/01/2025','DD/MM/YYYY')
AND S.D8DEB <  TO_DATE('01/01/2026','DD/MM/YYYY')
GROUP BY R.CDURM_P
ORDER BY R.CDURM_P;
