SELECT CATEGORIE,
       DUREE,
       COUNT(DISTINCT NOIP) AS NB_PATIENTS
FROM
(
    SELECT DISTINCT
           RSM.NOIP,
           CASE
             WHEN RSS.DUSEJ <= 3 THEN '3_JOURS_ET_MOINS'
             WHEN RSS.DUSEJ IN (4,5) THEN '4_OU_5_JOURS'
             ELSE 'AUTRE_DUREE'
           END AS DUREE,
           CASE
             WHEN (
                  RSM.CDDNC9 LIKE 'Z448%'
                  OR RSM.CDRELIE LIKE 'Z448%'
                  OR EXISTS (
                       SELECT 1
                       FROM DGN
                       WHERE DGN.KYRES = RSM.KYRES
                         AND DGN.KYCDDG LIKE 'Z448%'
                  )
             ) THEN 'DEBUT_POMPE_Z448'

             WHEN (
                  RSM.CDDNC9 LIKE 'Z469%'
                  OR RSM.CDRELIE LIKE 'Z469%'
                  OR EXISTS (
                       SELECT 1
                       FROM DGN
                       WHERE DGN.KYRES = RSM.KYRES
                         AND DGN.KYCDDG LIKE 'Z469%'
                  )
             ) THEN 'DEBUT_BOUCLE_FERMEE_Z469'

             WHEN (
                  (
                    RSM.CDDNC9 LIKE 'Z964%'
                    OR RSM.CDRELIE LIKE 'Z964%'
                    OR EXISTS (
                         SELECT 1
                         FROM DGN
                         WHERE DGN.KYRES = RSM.KYRES
                           AND DGN.KYCDDG LIKE 'Z964%'
                    )
                  )
                  AND NOT (
                    RSM.CDDNC9 LIKE 'Z448%'
                    OR RSM.CDRELIE LIKE 'Z448%'
                    OR EXISTS (
                         SELECT 1
                         FROM DGN
                         WHERE DGN.KYRES = RSM.KYRES
                           AND DGN.KYCDDG LIKE 'Z448%'
                    )
                    OR RSM.CDDNC9 LIKE 'Z469%'
                    OR RSM.CDRELIE LIKE 'Z469%'
                    OR EXISTS (
                         SELECT 1
                         FROM DGN
                         WHERE DGN.KYRES = RSM.KYRES
                           AND DGN.KYCDDG LIKE 'Z469%'
                    )
                  )
             ) THEN 'PORTEUR_POMPE_Z964_SANS_Z448_Z469'

             ELSE NULL
           END AS CATEGORIE
    FROM RSM, RUM, RSS
    WHERE RUM.NORES = RSM.KYRES
      AND RSS.KYRSS = RUM.KYRSS
      AND RSM.CDURM_P = '071S'
      AND RSS.D8DEB >= TO_DATE('01/01/2024','DD/MM/YYYY')
      AND RSS.D8DEB <  TO_DATE('01/01/2025','DD/MM/YYYY')
)
WHERE CATEGORIE IS NOT NULL
GROUP BY CATEGORIE, DUREE
ORDER BY CATEGORIE, DUREE;






SELECT CATEGORIE,
       COUNT(DISTINCT NOIP) AS NB_PATIENTS
FROM
(
    SELECT DISTINCT
           RSM.NOIP,
           CASE
             WHEN (
                  RSM.CDDNC9 LIKE 'Z451%'
                  OR RSM.CDRELIE LIKE 'Z451%'
                  OR EXISTS (
                       SELECT 1
                       FROM DGN
                       WHERE DGN.KYRES = RSM.KYRES
                         AND DGN.KYCDDG LIKE 'Z451%'
                  )
             ) THEN 'HDJ_POMPE_COMPLET_Z451'

             WHEN (
                  (
                    RSM.CDDNC9 LIKE 'Z964%'
                    OR RSM.CDRELIE LIKE 'Z964%'
                    OR EXISTS (
                         SELECT 1
                         FROM DGN
                         WHERE DGN.KYRES = RSM.KYRES
                           AND DGN.KYCDDG LIKE 'Z964%'
                    )
                  )
                  AND
                  (
                    RSM.CDDNC9 LIKE 'Z092%'
                    OR RSM.CDRELIE LIKE 'Z092%'
                    OR EXISTS (
                         SELECT 1
                         FROM DGN
                         WHERE DGN.KYRES = RSM.KYRES
                           AND DGN.KYCDDG LIKE 'Z092%'
                    )
                  )
             ) THEN 'SOUS_POMPE_BILAN_Z964_Z092'

             WHEN (
                  RSM.CDDNC9 LIKE 'Z469%'
                  OR RSM.CDRELIE LIKE 'Z469%'
                  OR EXISTS (
                       SELECT 1
                       FROM DGN
                       WHERE DGN.KYRES = RSM.KYRES
                         AND DGN.KYCDDG LIKE 'Z469%'
                  )
             ) THEN 'INITIATION_BOUCLE_FERMEE_Z469'

             WHEN (
                  RSM.CDDNC9 LIKE 'Z468%'
                  OR RSM.CDRELIE LIKE 'Z468%'
                  OR EXISTS (
                       SELECT 1
                       FROM DGN
                       WHERE DGN.KYRES = RSM.KYRES
                         AND DGN.KYCDDG LIKE 'Z468%'
                  )
             ) THEN 'PORTEURS_BOUCLE_FERMEE_Z468'

             ELSE NULL
           END AS CATEGORIE
    FROM RSM
    WHERE RSM.UE = '979'
      AND RSM.D8EEUE >= TO_DATE('01/01/2024','DD/MM/YYYY')
      AND RSM.D8EEUE <  TO_DATE('01/01/2025','DD/MM/YYYY')
)
WHERE CATEGORIE IS NOT NULL
GROUP BY CATEGORIE
ORDER BY CATEGORIE;
