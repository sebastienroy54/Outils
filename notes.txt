def trouver_header(df, liste_colonnes):
    import unicodedata

    def normaliser(s):
        if not isinstance(s, str):
            return ""
        s = s.strip().lower()
        s = ''.join(c for c in unicodedata.normalize('NFD', s)
                    if unicodedata.category(c) != 'Mn')
        return s

    liste_norm = [normaliser(c) for c in liste_colonnes]

    for i, row in df.iterrows():
        row_norm = [normaliser(x) for x in row.values]

        # On ne garde que les lignes qui contiennent au moins du texte
        if not any(len(x) > 0 for x in row_norm):
            continue

        # Vérification que les colonnes attendues sont présentes
        if all(col in row_norm for col in liste_norm):
            df.columns = row  # garde les noms tels qu'ils apparaissent
            df = df.drop(index=range(i+1)).reset_index(drop=True)
            return df

    raise ValueError(f"Impossible de trouver des colonnes : {liste_colonnes}")
