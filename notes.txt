def parse_excel_name(excel_filename: str):
    m = XLS_MCO_RE.match(excel_filename or "")
    if not m:
        return None
    return {
        "domain": "MCO",
        "site5": m.group(1),
        "yyyymm": m.group(2),
        "ext": m.group(3).lower(),
    }

XLS_MCO_RE = re.compile(
    r"^CORRE(?:S)?PONDANCE_UM_MCO_SITE_(\d{5})_APHP_(\d{6})\.(xls|xlsx)$",
    re.IGNORECASE,
)
