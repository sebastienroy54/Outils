SELECT
    r.NODA         AS numero_sejour,
    r.NOIP         AS ipp_patient,
    r.CDURM        AS code_service,
    d.KYCDDG       AS code_diagnostic,
    r.D8EEUE       AS date_entree
FROM RSM r
JOIN DGN d ON r.KYRES = d.KYRES
WHERE r.CDURM = '041'                 -- Chirurgie digestive
  AND d.KYCDDG = 'B965'               -- Code diagnostic recherch√©
  AND r.D8EEUE BETWEEN TO_DATE('2025-01-01','YYYY-MM-DD') AND SYSDATE
ORDER BY r.D8EEUE ASC;
