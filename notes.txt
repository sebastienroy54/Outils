SELECT DISTINCT RSM.NOIP
FROM RSM, DGN WHERE RSM.KYRES = DGN.KYRES(+)
AND
(
RSM.CDDNC9 in ('I330', 'I339') 
OR RSM.CDRELIE  in ('I330', 'I339') 
OR DGN.KYCDDG in ('I330', 'I339')
) 
AND RSM.CDURM_P in ('450', '451')
AND RSM.D8SOUE  >= to_date('01/01/2015','DD/MM/YYYY')
AND RSM.D8SOUE  < to_date('01/01/2025','DD/MM/YYYY')
;
spool off;
