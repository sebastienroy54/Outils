# Export XLSX avec format date dd/mm/yyyy
with pd.ExcelWriter(out_xlsx_path, engine="openpyxl") as writer:
    df_final.to_excel(writer, index=False, sheet_name="flat")
    ws = writer.sheets["flat"]

    headers = [cell.value for cell in ws[1]]
    date_cols = {"date_pose", "date_naissance", "nda_date_deb_match", "nda_date_fin_match"}
    date_col_indexes = [i + 1 for i, h in enumerate(headers) if h in date_cols]

    for col_idx in date_col_indexes:
        for row in range(2, ws.max_row + 1):
            cell = ws.cell(row=row, column=col_idx)
            if cell.value is not None:
                cell.number_format = "DD/MM/YYYY"
