SELECT DISTINCT
    RSM.NODA AS NDA,
    RSM.NOIP AS IPP,
    RSM.MDEEUE AS MODE_ENTREE_RSM,
    RSM.MDSOUE AS MODE_SORTIE_RSM,
    EPI.MDEEUE AS MODE_ENTREE_EPI,
    EPI.MDSOUE AS MODE_SORTIE_EPI,
    RSM.D8EEUE AS DATE_ENTREE,
    RSM.D8SOUE AS DATE_SORTIE
FROM
    RSM,
    EPI
WHERE
    RSM.KYRES = EPI.NORES
    AND EPI.CHPPMSI = 'SSR'
    AND RSM.D8SOUE >= TO_DATE('01/03/2026','DD/MM/YYYY')
    AND RSM.D8SOUE <  TO_DATE('01/04/2026','DD/MM/YYYY')
    AND ROWNUM <= 50
ORDER BY
    RSM.D8SOUE;
