SELECT DISTINCT 
    S.KYNOIP AS IPP,
    R.NODA AS NDA,
    R.CDGHM AS GHM,
    R.DUSEJ AS duree_sejour
FROM SDO S, RSS R
WHERE S.KYNODA = R.NODA
	AND S.KYNOIP IN ('liste')
	AND R.D8DEB >= TO_DATE('01/01/2025','DD/MM/YYYY') 
    AND R.D8DEB <= SYSDATE
	AND R.DUSEJ >= 4
	AND SUBSTR(R.CDGHM, -1) IN ('1','2')
ORDER BY R.NODA;
