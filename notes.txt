                RM.NODA NDA_MERE,
                RM.KYRES KYRES_MERE,
                RM.D8EEUE ENTREE_MERE,
                RM.D8SOUE SORTIE_MERE,
                RM.CDDNC9 DP_MERE,
                SB.KYNOIP IPP_BEBE,
                SB.KYNODA NDA_BEBE,
                SB.DANAIS DANAIS_BEBE,
                RB.KYRES KYRES_BEBE,
                RB.D8EEUE ENTREE_BEBE,

     RSM RB
